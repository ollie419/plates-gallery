import type { NextApiRequest, NextApiResponse } from 'next' export default async function handler(req: NextApiRequest, res: NextApiResponse) { if (req.method !== 'POST') return res.status(405).json({ error: 'Method not allowed' }) if (req.headers['x-api-key'] !== process.env.UPLOAD_API_KEY) return res.status(401).json({ error: 'Unauthorized' }) const { email, galleryUrl, eventName } = req.body if (!email || !galleryUrl) return res.status(400).json({ error: 'Missing email or galleryUrl' }) const response = await fetch('https://api.resend.com/emails', { method: 'POST', headers: { 'Authorization': `Bearer ${process.env.RESEND_API_KEY}`, 'Content-Type': 'application/json', }, body: JSON.stringify({ from: 'onboarding@resend.dev', to: email, subject: `Your photos from ${eventName ?? 'MOMNT'}`, html: `

Your photos are ready

Tap the link below to view and download your images.

View your photos

${galleryUrl}

`, }), }) if (!response.ok) { const error = await response.json() console.error('Resend error:', error) return res.status(500).json({ error: 'Failed to send email' }) } return res.status(200).json({ success: true }) }
